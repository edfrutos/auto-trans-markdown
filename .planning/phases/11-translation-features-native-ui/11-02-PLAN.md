### 1. Implementar `application(_:open:)` en `AppDelegate.swift`

```swift

func application(_ application: NSApplication, open urls: [URL]) {
    // Solo procesar archivos .md; ignorar el resto silenciosamente.
    let markdownURLs = urls.filter { $0.pathExtension.lowercased() == "md" }
    guard let first = markdownURLs.first else { return }

    do {
        let content = try String(contentsOf: first, encoding: .utf8)
        NotificationCenter.default.post(
            name: WebView.openMarkdownNotification,
            object: content
        )
    } catch {
        let alert = NSAlert()
        alert.messageText = "No se pudo abrir el archivo"
        alert.informativeText = "\(first.lastPathComponent): \(error.localizedDescription)"
        alert.alertStyle = .warning
        alert.runModal()
    }
}
```
